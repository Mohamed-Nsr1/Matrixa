function loadData() { 
    try { 
        const raw = localStorage.getItem(STORAGE_KEY); 
        if (!raw) return initializeState(); 
        let data = JSON.parse(raw);
        if (!data.planner) data.planner = initialPlanner;

        data.subjects.forEach(s => {
            if (s.notes === undefined) s.notes = [];

                    if (l.notes === undefined) l.notes = [];
                });
            });
        });
        return data;

function FocusSessionView({ task, onUpdateTask, onFinish, appState, toggleLessonFlag }) {
    const [secondsLeft, setSecondsLeft] = useState(task.remainingDuration === undefined ? task.duration * 60 : task.remainingDuration);
    const [isPaused, setIsPaused] = useState(true);

    useEffect(() => {
        if (isPaused || secondsLeft <= 0) return;
        const timerId = setInterval(() => setSecondsLeft(s => s > 0 ? s - 1 : 0), 1000);
        return () => clearInterval(timerId);
    }, [isPaused, secondsLeft]);

    const handleFinishSession = () => {
        const dayIds = ['sunday', 'monday', 'tuesday', 'wednesday', 'thursday', 'friday', 'saturday'];
        const todayId = dayIds[new Date().getDay()];
        onUpdateTask(todayId, { ...task, remainingDuration: secondsLeft });
        onFinish();
    };

    const minutes = Math.floor(secondsLeft / 60);
    const seconds = secondsLeft % 60;
    
    const lesson = useMemo(() => {
        for (const subject of appState.subjects) {
            for (const unit of subject.units) {
                const found = unit.lessons.find(l => l.id === task.lessonId);
                if (found) return found;
            }
        }
        return null;
    }, [appState.subjects, task.lessonId]);

    return (
        <div className="fixed inset-0 bg-[#0f172a] text-white flex flex-col items-center justify-center p-6 z-50">
